import random

def hangman():
    # List of predefined words
    words = ["apple", "banana", "orange", "grape", "mango"]
    word = random.choice(words)   # Pick a random word
    guessed = ["_"] * len(word)   # Hidden word with underscores
    attempts = 6                  # Max incorrect guesses
    used_letters = []             # Store guessed letters

    print("ğŸ® Welcome to Hangman!")
    print("Guess the word: ", " ".join(guessed))

    while attempts > 0 and "_" in guessed:
        guess = input("\nEnter a letter: ").lower()

        if not guess.isalpha() or len(guess) != 1:
            print("âŒ Please enter a single alphabet.")
            continue

        if guess in used_letters:
            print("âš  You already guessed that letter.")
            continue

        used_letters.append(guess)

        if guess in word:
            print("âœ… Correct guess!")
            for i in range(len(word)):
                if word[i] == guess:
                    guessed[i] = guess
        else:
            attempts -= 1
            print(f"âŒ Wrong guess! Attempts left: {attempts}")

        print("Word: ", " ".join(guessed))
        print("Used letters:", ", ".join(used_letters))

    # Game result
    if "_" not in guessed:
        print("\nğŸ‰ Congratulations! You guessed the word:", word)
    else:
        print("\nğŸ’€ Game Over! The word was:", word)

# Run the game
hangman()
